  <script>
    let duration = 5 * 60;
    let remainingTime = duration;
    let timerInterval = null;
    let playingSound = false;
    const timerDisplay = document.getElementById('timer');
    const endSound = document.getElementById('endSound');

    function updateDisplay() {
      const minutes = Math.floor(remainingTime / 60).toString().padStart(2, '0');
      const seconds = (remainingTime % 60).toString().padStart(2, '0');
      timerDisplay.textContent = `${minutes}:${seconds}`;
    }

    function startTimer() {
      if (timerInterval) return;
      timerDisplay.classList.add('running');
      timerInterval = setInterval(() => {
        if (remainingTime > 0) {
          remainingTime--;
          updateDisplay();
        } else {
          clearInterval(timerInterval);
          timerInterval = null;
          timerDisplay.classList.remove('running');
          timerDisplay.classList.add('glow');
          playSoundThreeTimes();
        }
      }, 1000);
    }

    function pauseTimer() {
      clearInterval(timerInterval);
      timerInterval = null;
      timerDisplay.classList.remove('running');
    }

    function resetTimer() {
      clearInterval(timerInterval);
      timerInterval = null;
      remainingTime = duration;
      updateDisplay();
      timerDisplay.classList.remove('running', 'glow');
      stopSound();
    }

    function setTime(minutes) {
      pauseTimer();
      duration = minutes * 60;
      remainingTime = duration;
      updateDisplay();
      timerDisplay.classList.remove('glow');
      stopSound();
    }

    function setCustomTime() {
      const input = document.getElementById('customTime').value.trim();
      const match = input.match(/^(\d{1,2}):(\d{2})$/);
      if (match) {
        let min = parseInt(match[1]);
        let sec = parseInt(match[2]);
        if (!isNaN(min) && !isNaN(sec) && sec < 60) {
          duration = (min * 60) + sec;
          remainingTime = duration;
          pauseTimer();
          updateDisplay();
          timerDisplay.classList.remove('glow');
          stopSound();
        } else {
          alert("Invalid time format.");
        }
      } else {
        alert("Please enter time in MM:SS format.");
      }
    }

    function add30Seconds() {
      remainingTime += 30;
      updateDisplay();
    }

    function playSoundThreeTimes() {
      let count = 0;
      playingSound = true;
      endSound.volume = 1.0;
      function playNext() {
        if (count < 3 && playingSound) {
          endSound.currentTime = 0;
          endSound.play()
            .then(() => {
              count++;
            })
            .catch(err => console.log("Audio play prevented:", err));
          endSound.onended = () => {
            playNext();
          };
        }
      }
      playNext();
    }

    function stopSound() {
      playingSound = false;
      endSound.pause();
      endSound.currentTime = 0;
    }

    function toggleSpotify() {
      const panel = document.getElementById('spotifyPanel');
      panel.classList.toggle('open');
    }

    updateDisplay();
  </script>
